### 当前配置文件（2021年3月20日） ###

#### `~/.emacs` ####

``` emacs-lisp
;;; ----- basics-----

;; basic face
(tool-bar-mode 0)
(menu-bar-mode 0)
(scroll-bar-mode 0)
(global-linum-mode 1)

;; English Font
(set-face-attribute
'default nil :font "Ubuntu Mono 16")

;; Chinese Font
(dolist (charset '(kana han symbol cjk-misc bopomofo))
(set-fontset-font (frame-parameter nil 'font)
charset
(font-spec :family "STZhongSong")))

;; space indent
(setq tab-width 4)
(setq-default indent-tabs-mode nil)

;; shut welcome page
(setq inhibit-splash-screen 1)


;; ----- packages-----


;; source
(require 'package)
(add-to-list 'package-archives
             '("melpa-stable" . "https://stable.melpa.org/packages/") t)
(add-to-list 'package-archives
             '("melpa" . "https://melpa.org/packages/") t)
(package-initialize)

(require 'use-package) ;; use-package

;; theme
(use-package nord-theme
  :ensure t
  :config (load-theme 'nord t))
;; leuven/dark, peacock, dracula, monokai, zenburn,
;; sanityinc-tomorrow-eighties, nord

;; flycheck
(use-package flycheck
  :ensure t
  :init (global-flycheck-mode))

;; orgmode
(require 'org)
(setq org-startup-indented t)
(global-set-key "\C-ca" 'org-agenda)
(setq org-log-done t) ;; close timestamp
(setq org-src-fontify-natively t) ;; code block highlight
(setq org-todo-keywords
      '((sequence "TODO(t)" "DOING(i@/!)" "|"
                  "DONE(d!)" "PASS(p@/!)" "CANCELLED(c@/!)")))

;; markdown mode
(use-package markdown-mode
  :ensure t
  :commands (markdown-mode gfm-mode)
  :mode (("README\.md\'" . gfm-mode)
         ("\.md\'" . markdown-mode)
         ("\.markdown\'" . markdown-mode))
  :init (setq markdown-command "multimarkdown"))

;; magit
(use-package magit
  :ensure t
  :bind (("C-x g" . magit-status)
         ("C-x M-g" . magit-dispatch)))

;; ein
(use-package ein
  :ensure t)

;; auto-complete
(use-package auto-complete
  :ensure t
  :config (ac-config-default))

;; ido-mode
(use-package ido
  :ensure t
  :init
  (setq ido-enable-flex-matching t)
  (setq ido-everywhere t)
  (setq ido-ignore-extensions t)
  (setq ido-auto-merge-delay-time 30)
  :config
  (ido-mode t))

(use-package editorconfig
  :ensure t
  :config
  (editorconfig-mode 1))

(provide '.emacs)
;;; .emacs ends here
```

### 其他 & 参考资源 ###

- [flycheck](https://www.flycheck.org/en/latest/)
  - 使用Clang, 效果更佳.
  - 使用gcc[可能出现的问题](https://github.com/flycheck/flycheck/issues/1578)是gcc的中文提示, 曾通过修改环境变量`LC_MESSAGES=en_US.UTF-8`解决.
- [auto-complete](https://github.com/auto-complete/auto-complete)
  - `(ac-config-default)`启用默认设置即可.
- [org-mode](https://orgmode.org/)
  - 注意检查是否能导出LaTeX.
  - 添加的文件头:
  ```org
  #+latex_header: \usepackage[UTF8]{ctex}
  #+latex_compiler: xelatex
  ```
  - 不再使用Minted高亮, 用自带的List高亮, 避免更多问题.
- [markdown-mode](https://jblevins.org/projects/markdown-mode/)
- [ein](https://github.com/millejoh/emacs-ipython-notebook)
- [magit](https://magit.vc/)
